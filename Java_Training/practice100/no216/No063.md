==ではなくequalsで比較する・・・文字型を==で比較すると参照先の比較をされるので値が同じでもfalseが返ってきてしまう



public class Knock063 {
    public int check1(String target) {
        if (target.equals("value1") {
            return 0;
        } else if (target.equals("value2")) {
            return 1;
        }
        return 2;
    }

    public int check2(String target) {
        if (target.equals("value1")) {
            return 0;
        } else if (target.equals("value2")) {
            return 1;
        }
        return 2;
    }
}
